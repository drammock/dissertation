%===== MOTIVATION =====%
\section{Motivation}

\subsection{Why study intelligibility?}
\begin{frame}{Why study intelligibility?}
	\begin{itm}
	\item<1-3> \emph{What is the same across talkers?}
		\begin{itm}
		\item Can tell us about the brain’s speech processing system
			\begin{itm}
			\item One example: target–masker similarity
			\end{itm}
		\end{itm}
	\item[]
	\item<2-3> \emph{What is different across talkers?}
		\begin{itm}
		\item Applications to assistive hearing device design (acoustics, signal processing)
		\item Applications to speech therapy, pedagogy (articulation)
		\end{itm}
	\item[]
	\item<3> \emph{What is different across groups?}
		\begin{itm}
		\item Cross-linguistic differences in speech perception
			\begin{itm}
			\item Potential for insight into “active” phonological features
			\end{itm}
		\end{itm}
	\end{itm}
\end{frame}

\subsection{Why study prosody?}
\begin{frame}{Why study prosody?}
	\begin{itm}
	\item<1-3> Hearing aids: signal processing for prosodic changes easier than segmental changes
	\item[]
	\item<2-3> Natural distinction: segmental \vs\ supersegmental
		\begin{itm}
		\item Duration, \fo, and intensity are not primary cues to phonemic distinctions in English as spoken here
		\end{itm}
	\item[]
	\item<3> Bottom-up approach to intelligibility using acoustic predictors has yielded conflicting results
	\end{itm}
\end{frame}

\subsection{What determines intelligibility?}
\begin{frame}{What determines intelligibility? (1 of 2)}{}
{\strong Properties of the talker \& listener}
	\begin{itm}
	\item Speech style
		\begin{itm}
		\item Clear speech, Lombard speech, parentese, sociolinguistic register, \etc.
		\end{itm}
	\item[]
	\item Listener familiarity with language/dialect
	\item[]
	\item Intrinsic individual differences?
	\item[]
	\item Intrinsic gender differences?
	\end{itm}
\end{frame}

\begin{frame}{What determines intelligibility? (2 of 2)}{}
{\strong Acoustic properties of the speech}
	\begin{itm}
	\item Duration \& speech rate
		\begin{itm}
		\item \emph{Long vowels, systematic pauses} associated with intelligibility\footnote[frame]{\citet*{BondMoore1994}.}
		\item No correlation between intelligibility and \emph{mean sentence duration}\footnote[frame]{\citet*{BradlowEtAl1996}.}
%		\item Clear speech with pauses removed doesn’t get less intelligible\footnote[frame]{Choi1987 - MIT bachelor’s project}
		\item Clear speech advantage can be achieved without changing speech rate\footnote[frame]{\citet*{KrauseBraida2002}.} %norm:  (187 \ac{wpm}), fast: (218 \ac{wpm})
		\item \emph{Equivocal results regarding speech rate} 
		\end{itm}
%	\item[]
	\item Larger vowel space size is associated with intelligibility
		\begin{itm}
		\item Greater F2 and/or F1 range\footnotemark[1] \footnotemark[2] \footnote[frame]{\citet*{HazanMarkham2004}.}
		\item Greater area of polygon formed by vowel means\footnotemark[2] \footnote[frame]{\citet*{Neel2008}} \footnote[frame]{\citet*{McCloyEtAl2012}}
		\item Greater area of convex polygonal hull encompassing all vowel tokens\footnote[frame]{\citet*{McCloyEtAl2013}.}
		\item \emph{No consensus on best way to measure}
		\end{itm}
%	\item[]
	\end{itm}
\end{frame}

\begin{frame}{}
	\begin{itm}
	\item Pitch
		\begin{itm}
		\item Mean pitch unrelated to intelligibility
		\item Larger pitch range associated with intelligibility\footnote[frame]{\citet*{BradlowEtAl1996}},\footnote[frame]{\citet*{McCloyEtAl2012}}
		\end{itm}
	\end{itm}
\end{frame}

\subsection{General approach}
\begin{frame}{General approach}
	\begin{itm}
	\item The split between top-down \vs\ bottom-up modeling intelligibility is not perfect:\\ \vspace{-\baselineskip}
	\begin{columns}[t]
		\begin{column}{0.4\textwidth}
			\begin{itm}
			\item Vowel quality reduction affected by phrasal accent
			\item Consonant reduction (spirantization, glottalization, flapping) affected by prosody
			\end{itm}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{itm}
			\item Duration is secondary cue to phonemic distinctions
			\item Pitch is (subtly) affected by tongue movements \& vocal tract constrictions (usually at the edges of vowels)
			\end{itm}
		\end{column}
	\end{columns}
	\item[]
	\item<3> Nevertheless, those are relatively small perturbations compared to:\\ \vspace{-\baselineskip}
	\begin{columns}[t]
		\begin{column}{0.4\textwidth}
			\begin{itm}
			\item Lexically-motivated vowel reduction (patterns of word stress)
			\item Phonotactically motivated consonant reduction
			\end{itm}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{itm}
			\item Duration changes due to phrasal stress, speech rate, final lengthening, \etc.
			\item Sentence-level pitch changes
			\end{itm}
		\end{column}
	\end{columns}

\begin{frame}{Why use resynthesis?}
	\begin{itm}
	\item Advantages: 
		\begin{itm}
		\item Less work than parametric synthesis
		\item Natural-sounding voices
		\end{itm}
	\item[]
	\item Disadvantages: 
		\begin{itm}
		\item Risk of distortion
		\item Real voices!
		\end{itm}
	\end{itm}
\end{frame}

\begin{frame}{Stimulus design}
	\begin{itm}
		\item Careful recordings of parallel sentences (head mounted mics, low noise floor, coached to standardize prosody)
		\item[]
		\item Five \ac{pnw} male talkers included in corpus; three selected for resynthesis (highest, lowest, \& midpoint intelligibility)
		\item[]
		\item Duration, intensity \& pitch swapped between all possible pairs of talkers
	\end{itm}
\end{frame}
