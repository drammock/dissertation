\section{Motivation}

\subsection{Why study intelligibility?}
\begin{frame}{Why study intelligibility?}
	\begin{itm}
	\item<1-3> \emph{What is the same across talkers?}
		\begin{itm}
		\item Can tell us about the brain’s speech processing system
			\begin{itm}
			\item One example: target–masker similarity
			\end{itm}
		\end{itm}
	\item[]
	\item<2-3> \emph{What is different across talkers?}
		\begin{itm}
		\item Applications to assistive hearing device design (acoustics, signal processing)
		\item Applications to speech therapy, pedagogy (articulation)
		\end{itm}
	\item[]
	\item<3> \emph{What is different across groups?}
		\begin{itm}
		\item Cross-linguistic differences in speech perception
			\begin{itm}
			\item Potential for insight into “active” phonological features
			\end{itm}
		\end{itm}
	\end{itm}
\end{frame}

\subsection{Why study prosody?}
\begin{frame}{Why study prosody?}
	\begin{itm}
	\item<1-3> Hearing aids: signal processing for prosodic changes easier than segmental changes
	\item[]
	\item<2-3> Natural distinction: segmental \vs\ supersegmental
		\begin{itm}
		\item Duration, \fo, and intensity are not primary cues to phonemic distinctions in English as spoken here
		\end{itm}
	\item[]
	\item<3> Bottom-up approach to intelligibility using acoustic predictors has yielded conflicting results
	\end{itm}
\end{frame}

\subsection{What determines intelligibility?}
\begin{frame}{What determines intelligibility? (1 of 2)}
Properties of the talker \& listener:
	\begin{itm}
	\item Speech style
		\begin{itm}
		\item Clear speech, Lombard speech, parentese, sociolinguistic register, \etc.
		\end{itm}
	\item[]
	\item Listener familiarity with language/dialect
	\item[]
	\item Intrinsic individual differences?
	\item[]
	\item Intrinsic gender differences?
	\end{itm}
\end{frame}

\begin{frame}{What determines intelligibility? (2 of 2)}
Acoustic properties of the speech:
	\begin{itm}
	\item Speech rate
		\begin{itm}
		\item Long vowels, systematic pauses associated with intelligible speech \citep{BondMoore1994}
		\item No correlation between intelligibility and mean sentence duration \citep{BradlowEtAl1996}
		\item Clear speech advantage can be achieved at high speech rates \citep{KrauseBraida2002}
		\end{itm}
	\item[]
	\item Vowel space size \citep{BondMoore1994, BradlowEtAl1996, HazanMarkham2004, Neel2008, McCloyEtAl2012}
		\begin{itm}
		\item No clear agreement on how to measure this
		\end{itm}
	\item[]
	\item Pitch range?
	\end{itm}
\end{frame}

%===== METHODS =====%
\section{Methodology}

\begin{frame}{General approach}
	\begin{itm}
		\item<1-3> Top-down \vs\ bottom-up
		\begin{itm}
			\item<2-3> Not a perfect split:
			\begin{itm}
				\item Vowel reduction (formant values) affected by phrasal accent
				\item Consonant reduction (spirantization, glottalization, flapping) affected by prosody
				\item Duration is secondary cue to phonemic distinctions
				\item Pitch is (subtly) affected by vocal tract constrictions (\ie, consonants)
			\end{itm}
		\end{itm}
		\item[]
		\item<3> Why use resynthesis?
		\begin{itm}
			\item Advantage: Less work than parametric synthesis; natural-sounding voices
			\item Disadvantage: risk of distortion; real voices!
		\end{itm}
	\end{itm}
\end{frame}

\begin{frame}{Stimulus design}
	\begin{itm}
		\item Careful recordings of parallel sentences (head mounted mics, low noise floor, coached to standardize prosody)
		\item[]
		\item Five \ac{pnw} male talkers included in corpus; three selected for resynthesis (highest, lowest, \& midpoint intelligibility)
		\item[]
		\item Duration, intensity \& pitch swapped between all possible pairs of talkers
	\end{itm}
\end{frame}
